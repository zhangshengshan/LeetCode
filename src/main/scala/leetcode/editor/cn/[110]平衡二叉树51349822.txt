object Solution {
  def getHeight(left: TreeNode, right: TreeNode): Boolean = {
    return math.abs(maxDepth(left), maxDepth(right)) <= 1
  }

  def maxDepth(root: TreeNode): Int = {
    if (root == null) return 0
    return 1 + math.max(maxDepth(root.left), maxDepth(root.right))
  }

  def isBalanced(root: TreeNode): Boolean = {
    if (root == null) return null
    return getHeight(root.left, root.right) && isBalanced(root.left) && isBalanced(root.right)
  }
}
